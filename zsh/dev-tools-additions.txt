# --- REQUIRED: fnm (Fast Node Manager) ---
# Lets your terminal find and use fnm, and automatically switches Node.js
# versions when you move between project folders. Without this line, the
# `fnm`, `node`, and `npm` commands won't work even after installing them.
eval "$(fnm env --use-on-cd --shell zsh)"


# --- NPM GLOBAL TOOLBOX ---
# Skip this unless you've read the "npm Global Toolbox" section in the main
# README and decided you want it. It keeps globally installed npm tools
# (like Claude Code) in one fixed folder, so they don't disappear when you
# switch Node.js versions with fnm.
#
# Unlike some blocks in this file, this one takes effect the moment it
# loads: npm starts installing global packages into ~/.npm-global right
# away, even if you haven't finished the README's setup steps yet. Delete
# this block if you're not ready for that.
#
# This block must go BELOW the fnm line above, and below any other lines
# in this file that change your PATH.
export NPM_CONFIG_PREFIX="$HOME/.npm-global"
path=("$HOME/.npm-global/bin" $path)
export NODE_PATH="$HOME/.npm-global/lib/node_modules${NODE_PATH:+:$NODE_PATH}"
