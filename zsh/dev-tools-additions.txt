# --- REQUIRED: fnm (Fast Node Manager) ---
# Lets your terminal find and use fnm, and automatically switches Node.js
# versions when you move between project folders. Without this line, the
# `fnm`, `node`, and `npm` commands won't work even after installing them.
eval "$(fnm env --use-on-cd --shell zsh)"


# --- OPTIONAL / ADVANCED: npm Global Toolbox ---
# Skip this block unless you've read the "npm Global Toolbox" section in
# this repo's README and decided you want it. It keeps globally installed
# npm tools (like Claude Code) in one fixed folder, so they don't disappear
# when you switch Node.js versions with fnm.
#
# Important: this block must go BELOW the fnm line above, and below any
# other lines in this file that change your PATH.

# Point npm's global installs at a folder that doesn't move
export NPM_CONFIG_PREFIX="$HOME/.npm-global"
path=("$HOME/.npm-global/bin" $path)

# Help Node find packages installed in that folder
export NODE_PATH="$HOME/.npm-global/lib/node_modules${NODE_PATH:+:$NODE_PATH}"
